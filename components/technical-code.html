JavaScript 
// Initialize the AHP client
const ahp = new AHPClient({
  apiKey: "your-api-key",
  version: "2.0"
});

// Process a human message
async function processMessage(message) {
  const response = await ahp.process({
    message,
    model: "gpt-4",
    context: {
      user: currentUser,
      conversation: conversationHistory
    }
  });

  // Handle the AI response
  if (response.intent.primary === "query") {
    displayResponse(response.ai_response);
  } else if (response.intent.primary === "action") {
    performAction(response.action_details);
  }

  return response;
}
Note: The JavaScript client supports both browser and Node.js environments. For browser usage, we recommend using our CDN-hosted version for optimal performance.
